     | python3 -m json.tool

  - last_action_result: "success"
  - reward > 0 (rescuing casualties)
  - reward_breakdown shows r_rescue > 0.0

# 3c. Send supplies to Zone A
curl -s -X POST $BASE/step \
     -H "Content-Type: application/json" \
     -d "{\"session_id\":\"$SID\",\"action\":{\"action\":\"send_supplies\",\"to_zone\":\"A\",\"units\":5}}" \
     | python3 -m json.tool

Expected:
  - last_action_result: "success"
  - reward_breakdown shows r_supply > 0.0

# 3d. View full internal state (grader view — includes hidden fields)
curl -s $BASE/state/$SID | python3 -m json.tool

Expected: zones array includes is_false_sos and casualties_critical fields
  (these are hidden from agent but visible to grader)

# 3e. Keep stepping until done=true (deploy teams / send supplies each step)
#     Repeat the deploy_team and send_supplies calls from 3b/3c.
#     When done=true appears in the response, proceed to 3f.

# 3f. Run the grader on the completed episode
curl -s $BASE/state/$SID > /tmp/final_state.json
cat /tmp/final_state.json | python3 -c "
import json, sys
state = json.load(sys.stdin)
print('Event log length:', len(state.get('event_log', [])))
print('Cumulative reward:', state.get('cumulative_reward'))
print('Done:', state.get('done'))
"

What it proves: Complete episode lifecycle works end-to-end.

STEP 5 — TASK 3 NOVEL MECHANICS (Hard — false SOS + dam break)
================================================================

# 5a. Reset task_3
curl -s -X POST $BASE/reset \
     -H "Content-Type: application/json" \
     -d '{"task_id":"task_3"}' | python3 -m json.tool

Note session_id → SID3="<id>"

# 5b. Check hidden fields — zones H, I, J are false SOS zones
curl -s $BASE/state/$SID3 | python3 -c "
import json, sys
state = json.load(sys.stdin)
for z in state['zones']:
    if z['is_false_sos']:
        print(f'FALSE SOS zone: {z[\"zone_id\"]} | casualties_total={z[\"casualties_total\"]} | should be 0')
"

Expected: Zones H, I, J printed with casualties_total=0

# 5c. Verify false SOS zones LOOK real to agent (sos_active=true in observation)
#     The observation returned by /reset shows sos_active but NOT is_false_sos
#     Look at the observation from step 5a — zones H, I, J should show sos_active: true

# 5d. Deploy to a false SOS zone (Zone H) to verify penalty fires
curl -s -X POST $BASE/step \
     -H "Content-Type: application/json" \
     -d "{\"session_id\":\"$SID3\",\"action\":{\"action\":\"send_supplies\",\"to_zone\":\"H\",\"units\":2}}" \
     | python3 -m json.tool

Expected:
  - last_action_result: "success" (env doesn't reject false SOS — agent must detect it)
  - reward_breakdown shows p_false_sos > 0.0 (penalty applied)

# 5e. Trigger dam break — step 7 times total to reach dam break
#     (Dam break fires during the 8th step call)
#     Do 6 more steps (wait is fine, we already did 1 step above):

for i in 1 2 3 4 5 6; do
  curl -s -X POST $BASE/step \
       -H "Content-Type: application/json" \
       -d "{\"session_id\":\"$SID3\",\"action\":{\"action\":\"wait\"}}" \
       | python3 -c "import json,sys; r=json.load(sys.stdin); print(f'step {r[\"observation\"][\"step_number\"]}: reward={r[\"reward\"]:.4f}')"
done

# 5f. Check Zone E casualties AFTER dam break (should have jumped significantly)
curl -s $BASE/state/$SID3 | python3 -c "
import json, sys
state = json.load(sys.stdin)
for z in state['zones']:
    if z['zone_id'] == 'E':
        remaining = z['casualties_total'] - z['casualties_rescued']
        print(f'Zone E casualties_total={z[\"casualties_total\"]} rescued={z[\"casualties_rescued\"]} remaining={remaining}')
print('Events:', [e.get(\"type\") for e in state.get(\"event_log\", [])])
"

Expected:
  - Zone E casualties_total significantly higher than initial (dam break added ~60 casualties)
  - event_log contains a "dam_break" event type

What it proves: False SOS mechanics, dam break cascading failure, all 3 novel mechanics work.

STEP 8 — GRADER ENDPOINT
================================================================

# 8a. Run a full task_1 greedy episode and grade it
curl -s -X POST $BASE/reset \
     -H "Content-Type: application/json" \
     -d '{"task_id":"task_1"}' | python3 -c "
import json, sys, subprocess, os
data = json.load(sys.stdin)
sid = data['session_id']
base = os.environ.get('BASE', 'http://localhost:7860')
print('Session:', sid)

# Greedy: deploy + supply every step
for i in range(10):
    r = subprocess.run(['curl','-s','-X','POST', f'{base}/step',
        '-H','Content-Type: application/json',
        '-d', json.dumps({'session_id': sid, 'action': {'action':'deploy_team','to_zone':'A','units':3}})],
        capture_output=True, text=True)
    resp = json.loads(r.stdout)
    if resp.get('done'):
        print('Episode done at step', resp['observation']['step_number'])
        break

# Grade it
state_r = subprocess.run(['curl','-s', f'{base}/state/{sid}'], capture_output=True, text=True)
state = json.loads(state_r.stdout)
grade_r = subprocess.run(['curl','-s','-X','POST', f'{base}/grader',
    '-H','Content-Type: application/json',
    '-d', json.dumps({'event_log': state['event_log'], 'final_state': state, 'task_id': 'task_1'})],
    capture_output=True, text=True)
print('Grader result:', grade_r.stdout)
"

Expected:
  - score in [0.0, 1.0]
  - breakdown object with all 8 dimensions scored

What it proves: /grader endpoint works, all 8 scoring dimensions active.
